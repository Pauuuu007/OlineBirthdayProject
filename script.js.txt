const cards = document.querySelectorAll(".card");
const popup = document.getElementById("popup");
const closeBtn = document.getElementById("closeBtn");
const quoteText = document.getElementById("quoteText");

const quotes = [
  "Hidup itu sederhana, kita yang membuatnya rumit.",
  "Jangan menyerah, hal besar butuh waktu.",
  "Bahagia bukan karena segalanya sempurna, tapi karena melihat yang sempurna dalam segalanya.",
  "Kesempatan tidak datang dua kali.",
  "Lakukan sekarang, besok belum tentu ada.",
  "Percaya diri adalah kunci kesuksesan.",
  "Hidup bukan menunggu badai reda, tapi belajar menari di tengah hujan.",
  "Kegagalan adalah bagian dari proses menuju sukses.",
  "Setiap hari adalah kesempatan baru.",
  "Kamu lebih kuat dari yang kamu kira."
];

// Klik kartu → munculkan popup dengan quotes acak
cards.forEach(card => {
  card.addEventListener("click", () => {
    const randomQuote = quotes[Math.floor(Math.random() * quotes.length)];
    quoteText.textContent = randomQuote;
    popup.classList.remove("hidden");
  });
});

// Tombol X → tutup popup
closeBtn.addEventListener("click", () => {
  popup.classList.add("hidden");
});
